if (!id) { document.getElementById('loginMsg').textContent = 'ID를 입력하세요'; return; } fetch(`apiauthenticateid=${encodeURIComponent(id)}`) .then(res = res.json().then(j = ({ok res.ok, ...j}))) .then(data = { if (data.authenticated) { currentId = id; document.getElementById('loginSection').style.display = 'none'; document.getElementById('examSection').style.display = 'block'; loadExams(); } else { document.getElementById('loginMsg').textContent = '인증되지 않은 ID입니다'; sel.innerHTML = ''; exams.forEach(e = { const opt = document.createElement('option'); opt.value = e; opt.textContent = e; sel.appendChild(opt); }); }) .catch(() = { document.getElementById('examMsg').textContent = '회차 목록 로드 실패';
